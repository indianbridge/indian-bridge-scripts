			 * Get excerpt of post content for display in popup view

			public function get_post_excerpt() {
				if (
					! isset( $this->post->post_excerpt ) ||
					empty( $this->post->post_excerpt )
				) {
					$text = strip_shortcodes( $this->post->post_content );
					$text = apply_filters('the_content', $text);
					$text = str_replace(']]>', ']]&gt;', $text);
					$excerpt_length = apply_filters('excerpt_length', 25);
					$excerpt_more = apply_filters('excerpt_more', ' ' . '[...]');
					$this->post->post_excerpt = wp_trim_words( $text, $excerpt_length, $excerpt_more );		

				}
				return $this->post->post_excerpt;
			}
